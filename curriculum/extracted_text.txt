import fitz  # PyMuPDF 
 
def parse_pymupdf(file_path): 
    doc = fitz.open(file_path) 
 
    titles = [] 
    text = [] 
 
    for page_num, page in enumerate(doc): 
        content = page.get_text() 
 
        if not content: 
            continue 
 
        lines = content.split("\n") 
 
        for line in lines: 
            line = line.strip() 
 
            if not line: 
                continue 
 
            #    Title detection (important) 
            if ( 
                line.isupper() or 
                "TEST REPORT" in line or 
                "BRAKE" in line 
            ): 
                titles.append(line) 
            else: 
                text.append(line) 
 
    return { 
        "titles": titles, 
        "text": text 
    }

import fitz 
 
def parse_pymupdf(file_path): 
    doc = fitz.open(file_path) 
 
    pages = [] 
 
    for i in range(doc.page_count): 
        page = doc.load_page(i) 
 
        text_lines = str( 
            page.get_text("text") or "" 
        ).splitlines() 
 
        cleaned = [] 
 
        for line in text_lines: 
            line = line.strip() 
 
            if line: 
                cleaned.append(line) 
 
        pages.append({ 
            "page": i, 
            "text": cleaned 
        }) 
 
    return pages

import pdfplumber 
 
def clean_row(row): 
    return [ 
        cell.strip() 
        for cell in row 
        if cell not in (None, "" , " ") 
    ] 
 
def extract_tables_pdf(file_path): 
    tables = [] 
 
    with pdfplumber.open(file_path) as pdf: 
 
        for page_num, page in enumerate(pdf.pages): 
 
            raw_tables = page.extract_tables() 
 
            if not raw_tables: 
                continue 
 
            for table_id, table in enumerate(raw_tables): 
 
                cleaned_table = [] 
 
                for row in table: 
                    cleaned = clean_row(row) 
 
                    if len(cleaned) >= 2: 
                        cleaned_table.append(cleaned) 
 
                if len(cleaned_table) < 2: 
                    continue 
 
                tables.append({ 
                    "page": page_num, 
                    "table_id": f"page_{page_num}table{table_id}" , 
                    "data": cleaned_table 
                }) 
 
                print(f"   Table extracted from page {page_num}") 
 
    return tables

def build_page_view(file_path):

    extracted_tables = extract_tables_pdf(file_path) 
 
    table_map = {} 
 
    for t in extracted_tables: 
        page = t["page"] 
 
        if page not in table_map: 
            table_map[page] = [] 
 
        table_map[page].append(t) 
 
    final_pages = [] 
 
    for p in parsed_pages: 
 
        page_num = p["page"] 
 
        final_pages.append({ 
            "page": page_num, 
            "text": p["text"], 
            "tables": table_map.get(page_num, []) 
        }) 
 
    return final_pages

import pdfplumber 
import pandas as pd 
 
def extract_tables_pdf(pdf_path): 
 
    tables = [] 
 
    with pdfplumber.open(pdf_path) as pdf: 
 
        for page_num, page in enumerate(pdf.pages): 
 
            raw_tables = page.extract_tables() 
 
            if not raw_tables: 
                continue 
 
            for table_id, table in enumerate(raw_tables): 
 
                df = pd.DataFrame(table) 
 
                #    remove fully empty rows 
                df = df.dropna(axis=0, how="all") 
 
                #    remove fully empty columns 
                df = df.dropna(axis=1, how="all") 
 
                #    preserve merged/split cells 
                df = df.fillna(None) 
 
                tables.append({ 
                    "page": page_num, 
                    "table_id": f"page_{page_num}table{table_id}" , 
                    "df": df 
                }) 
 
                print(f"   Table extracted from page {page_num}") 
 
    return tables

     app/parser/table_segmenter.py 
import pandas as pd 
 
def get_row_profile(row): 
 
    values = [ 
        str(x).strip() 
        for x in row 
    ] 
 
    numeric_count = 0 
    text_count = 0 
    empty_count = 0 
 
    for v in values: 
 
        if v.lower() in ["" , "nan" , "none"]: 
            empty_count += 1 
 
        elif any(c.isdigit() for c in v): 
            numeric_count += 1 
 
        else: 
            text_count += 1 
 
    populated = len(values) - empty_count 
 
    return { 
        "numeric": numeric_count, 
        "text": text_count, 
        "empty": empty_count, 
        "populated": populated 
    } 
 
   STEP 2 — GENERIC SEGMENTATION 
def segment_dataframe(df): 
 
    segments = [] 
 
    current = [] 
 
    previous_profile = None 
 
    for _, row in df.iterrows(): 
 
        row_values = list(row) 
 
        profile = get_row_profile(row_values) 
 
        #    detect structural discontinuity 
        split_condition = False 
 
        if previous_profile: 
 
            diff = abs( 
                profile["populated"] - 
                previous_profile["populated"] 
            ) 
 
            #    sudden structure shift 
            if diff >= 3: 
                split_condition = True 
 
        if split_condition and current: 
 
            segments.append( 
                pd.DataFrame(current) 
            ) 
 
            current = [] 
 
        current.append(row_values) 
 
        previous_profile = profile 
 
    if current: 
        segments.append( 
            pd.DataFrame(current) 
        ) 
 
    return segments

     app/parser/semantic_propagator.py 
import pandas as pd 
 
def propagate_semantic_rows(df): 
 
    if len(df.columns) == 0: 
        return df 
 
    first_col = df.iloc[:, 0] 
 
    propagated = [] 
 
    previous_value = None 
 
    for val in first_col: 
 
        val_str = str(val).strip() 
 
        #    semantic parent row 
        if ( 
            val_str.lower() not in 
            ["" , "nan" , "none"] 
        ): 
 
            previous_value = val 
 
            propagated.append(val) 
 
        else: 
 
            propagated.append(previous_value) 
 
    df.iloc[:, 0] = propagated 
 
    return df

import pandas as pd 
 
from app.parser.table_segmenter import ( 
    segment_dataframe 
) 
 
from app.parser.semantic_propagator import ( 
    propagate_semantic_rows 
) 
 
def normalize_tables(tables): 
 
    final_tables = [] 
 
    for t in tables: 
 
        df = t["df"] 
 
        #    clean only fully empty rows/cols 
        df = df.dropna(axis=0, how="all") 
        df = df.dropna(axis=1, how="all") 
 
        #    preserve structure 
        df = df.fillna(None) 
 
        #    segment logically 
        segments = segment_dataframe(df) 
 
        for idx, seg in enumerate(segments): 
 
            #    semantic propagation 
            seg = propagate_semantic_rows(seg) 
 
            final_tables.append({ 
                "table_id": 
                    f"{t['table_id']}segment{idx}" , 
 
                "page": t["page"], 
 
                "df": seg 
            }) 
 
    return final_tables

import pandas as pd 
 
def filter_useful_tables(tables): 
 
    useful = [] 
 
    for t in tables: 
 
        df = t["df"].copy() 
 
        #    normalize empty values 
        df = df.map( 
            lambda x: pd.NA 
            if str(x).strip().lower() 
            in ["nan" , "none" , ""] 
            else x 
        ) 
 
        #    remove fully empty columns 
        df = df.dropna(axis=1, how="all") 
 
        #    remove fully empty rows 
        df = df.dropna(axis=0, how="all") 
 
        rows, cols = df.shape 
 
        #    skip useless tiny tables 
        if rows <= 1: 
            continue 
 
        if cols <= 1: 
            continue 
 
        #    update cleaned dataframe 
        t["df"] = df 
 
        useful.append(t) 
 
    return useful

def dataframe_to_chunks(table): 
 
    chunks = [] 
 
    df = table["df"] 
 
    page = table["page"] 
 
    table_id = table["table_id"] 
 
    for idx, row in df.iterrows(): 
 
        values = [ 
            str(v) 
            for v in row.values 
        ] 
 
        text = " | " .join(values) 
 
        chunks.append({ 
            "id": f"{table_id}row{idx}" , 
            "text": text, 
            "metadata": { 
                "page": page, 
                "table_id": table_id 
            } 
        }) 
 
    return chunks

from app.memory.session_memory import ( 
    get_history 
) 
 
def rewrite_query(current_query): 
 
    history = get_history() 
 
    #    previous user queries 
    previous_queries = [ 
 
        h["content"] 
 
        for h in history 
 
        if h["role"] == "user" 
    ] 
 
    #    no history 
    if not previous_queries: 
 
        return current_query 
 
    previous_query = previous_queries[-1] 
 
    #    avoid duplicate expansion 
    if current_query.lower() in previous_query.lower(): 
 
        return current_query 
 
    #    contextual rewrite 
    rewritten = ( 
        previous_query 
        + " " 
        + current_query 
    ) 
 
    return rewritten

def deduplicate_context(results): 
 
    unique = [] 
 
    seen = set() 
 
    for r in results: 
 
        if isinstance(r, dict): 
 
            text = str( 
                r.get("document" , "") 
            ).strip() 
 
        else: 
 
            text = str(r).strip() 
 
        #    skip duplicates 
        if text in seen: 
            continue 
 
        seen.add(text) 
 
        unique.append(text) 
 
    return unique

from sentence_transformers import ( 
    CrossEncoder 
) 
 
reranker = CrossEncoder( 
    "BAAI/bge-reranker-base" 
) 
 
def rerank_results(query, documents): 
 
    pairs = [] 
 
    for doc in documents: 
        pairs.append([ 
            query, 
            doc["document"] 
        ]) 
 
    scores = reranker.predict(pairs) 
 
    ranked = [] 
 
    for doc, score in zip(documents, scores): 
 
        doc["rerank_score"] = float(score) 
 
        ranked.append(doc) 
 
    ranked = sorted( 
        ranked, 
        key=lambda x: x["rerank_score"], 
        reverse=True 
    ) 
 
    return ranked

    combined = semantic_docs + keyword_docs

    final_docs = list(unique.values())

    reranked = rerank_results( 
        query, 
        final_docs 
    ) 
 
    return reranked[:5]

import fitz  # PyMuPDF 
 
def parse_pymupdf(file_path): 
    doc = fitz.open(file_path) 
 
    titles = [] 
    text = [] 
 
    for page_num, page in enumerate(doc): 
        content = page.get_text() 
 
        if not content: 
            continue 
 
        lines = content.split("\n") 
 
        for line in lines: 
            line = line.strip() 
 
            if not line: 
                continue 
 
            #    Title detection (important) 
            if ( 
                line.isupper() or 
                "TEST REPORT" in line or 
                "BRAKE" in line 
            ): 
                titles.append(line) 
            else: 
                text.append(line) 
 
    return { 
        "titles": titles, 
        "text": text 
    }

        text_lines = str(

        pages.append({

        }) 
 
    return pages

import pdfplumber 
 
def clean_row(row): 
    return [ 
        cell.strip() 
        for cell in row 
        if cell not in (None, "" , " ") 
    ] 
 
def extract_tables_pdf(file_path): 
    tables = [] 
 
    with pdfplumber.open(file_path) as pdf: 
 
        for page_num, page in enumerate(pdf.pages): 
 
            raw_tables = page.extract_tables() 
 
            if not raw_tables: 
                continue 
 
            for table_id, table in enumerate(raw_tables): 
 
                cleaned_table = [] 
 
                for row in table: 
                    cleaned = clean_row(row) 
 
                    if len(cleaned) >= 2: 
                        cleaned_table.append(cleaned) 
 
                if len(cleaned_table) < 2: 
                    continue 
 
                tables.append({ 
                    "page": page_num, 
                    "table_id": f"page_{page_num}table{table_id}" , 
                    "data": cleaned_table 
                }) 
 
                print(f"   Table extracted from page {page_num}") 
 
    return tables

from app.parser.pymupdf_parser import parse_pymupdf 
from app.parser.pdfplumber_table import extract_tables_pdf 
 
def build_page_view(file_path): 
 
    parsed_pages = parse_pymupdf(file_path) 
 
    extracted_tables = extract_tables_pdf(file_path) 
 
    table_map = {} 
 
    for t in extracted_tables: 
        page = t["page"] 
 
        if page not in table_map: 
            table_map[page] = [] 
 
        table_map[page].append(t) 
 
    final_pages = [] 
 
    for p in parsed_pages: 
 
        page_num = p["page"] 
 
        final_pages.append({ 
            "page": page_num, 
            "text": p["text"], 
            "tables": table_map.get(page_num, []) 
        }) 
 
    return final_pages

import pdfplumber 
import pandas as pd 
 
def extract_tables_pdf(pdf_path): 
 
    tables = [] 
 
    with pdfplumber.open(pdf_path) as pdf: 
 
        for page_num, page in enumerate(pdf.pages): 
 
            raw_tables = page.extract_tables() 
 
            if not raw_tables: 
                continue 
 
            for table_id, table in enumerate(raw_tables): 
 
                df = pd.DataFrame(table) 
 
                #    remove fully empty rows 
                df = df.dropna(axis=0, how="all") 
 
                #    remove fully empty columns 
                df = df.dropna(axis=1, how="all") 
 
                #    preserve merged/split cells 
                df = df.fillna(None) 
 
                tables.append({ 
                    "page": page_num, 
                    "table_id": f"page_{page_num}table{table_id}" , 
                    "df": df 
                }) 
 
                print(f"   Table extracted from page {page_num}") 
 
    return tables

import fitz 
 
def parse_pymupdf(file_path): 
 
    doc = fitz.open(file_path) 
 
    pages = [] 
 
    for i in range(doc.page_count): 
 
        page = doc.load_page(i) 
 
        text_lines = str( 
            page.get_text("text") or "" 
        ).splitlines() 
 
        cleaned = [] 
 
        for line in text_lines: 
 
            line = line.strip() 
 
            if line: 
                cleaned.append(line) 
 
        pages.append({ 
            "page": i, 
            "text": cleaned 
        }) 
 
    return pages

import pandas as pd 
 
def get_row_profile(row): 
 
    values = [ 
        str(x).strip() 
        for x in row 
    ] 
 
    numeric_count = 0 
    text_count = 0 
    empty_count = 0 
 
    for v in values: 
 
        if v.lower() in ["" , "nan" , "none"]: 
            empty_count += 1 
 
        elif any(c.isdigit() for c in v): 
            numeric_count += 1 
 
        else: 
            text_count += 1 
 
    populated = len(values) - empty_count 
 
    return { 
        "numeric": numeric_count, 
        "text": text_count, 
        "empty": empty_count, 
        "populated": populated 
    } 
 
   STEP 2 — GENERIC SEGMENTATION 
def segment_dataframe(df): 
 
    segments = [] 
 
    current = [] 
 
    previous_profile = None 
 
    for _, row in df.iterrows(): 
 
        row_values = list(row) 
 
        profile = get_row_profile(row_values) 
 
        #    detect structural discontinuity 
        split_condition = False 
 
        if previous_profile: 
 
            diff = abs( 
                profile["populated"] - 
                previous_profile["populated"] 
            ) 
 
            #    sudden structure shift 
            if diff >= 3: 
                split_condition = True 
 
        if split_condition and current: 
 
            segments.append( 
                pd.DataFrame(current) 
            ) 
 
            current = [] 
 
        current.append(row_values) 
 
        previous_profile = profile 
 
    if current: 
        segments.append( 
            pd.DataFrame(current) 
        ) 
 
    return segments

import pandas as pd 
 
def propagate_semantic_rows(df): 
 
    if len(df.columns) == 0: 
        return df 
 
    first_col = df.iloc[:, 0] 
 
    propagated = [] 
 
    previous_value = None 
 
    for val in first_col: 
 
        val_str = str(val).strip() 
 
        #    semantic parent row 
        if ( 
            val_str.lower() not in 
            ["" , "nan" , "none"] 
        ): 
 
            previous_value = val 
 
            propagated.append(val) 
 
        else: 
 
            propagated.append(previous_value) 
 
    df.iloc[:, 0] = propagated 
 
    return df

def normalize_tables(tables): 
 
    final_tables = [] 
 
    for t in tables:

                "page": t["page"], 
 
                "df": seg 
            }) 
 
    return final_tables

import pandas as pd 
 
def filter_useful_tables(tables): 
 
    useful = [] 
 
    for t in tables: 
 
        df = t["df"].copy() 
 
        #    normalize empty values 
        df = df.map( 
            lambda x: pd.NA 
            if str(x).strip().lower() 
            in ["nan" , "none" , ""] 
            else x 
        ) 
 
        #    remove fully empty columns 
        df = df.dropna(axis=1, how="all") 
 
        #    remove fully empty rows 
        df = df.dropna(axis=0, how="all") 
 
        rows, cols = df.shape 
 
        #    skip useless tiny tables 
        if rows <= 1: 
            continue 
 
        if cols <= 1: 
            continue 
 
        #    update cleaned dataframe 
        t["df"] = df 
 
        useful.append(t) 
 
    return useful

def dataframe_to_chunks(table): 
 
    chunks = [] 
 
    df = table["df"] 
 
    page = table["page"] 
 
    table_id = table["table_id"] 
 
    for idx, row in df.iterrows(): 
 
        values = [ 
            str(v) 
            for v in row.values 
        ] 
 
        text = " | " .join(values) 
 
        chunks.append({ 
            "id": f"{table_id}row{idx}" , 
            "text": text, 
            "metadata": { 
                "page": page, 
                "table_id": table_id 
            } 
        }) 
 
    return chunks

def rewrite_query(current_query): 
 
    history = get_history() 
 
    #    previous user queries 
    previous_queries = [ 
 
        h["content"] 
 
        for h in history 
 
        if h["role"] == "user" 
    ] 
 
    #    no history 
    if not previous_queries: 
 
        return current_query 
 
    previous_query = previous_queries[-1] 
 
    #    avoid duplicate expansion 
    if current_query.lower() in previous_query.lower(): 
 
        return current_query 
 
    #    contextual rewrite 
    rewritten = ( 
        previous_query 
        + " " 
        + current_query 
    ) 
 
    return rewritten

def deduplicate_context(results): 
 
    unique = [] 
 
    seen = set() 
 
    for r in results: 
 
        if isinstance(r, dict): 
 
            text = str( 
                r.get("document" , "") 
            ).strip() 
 
        else: 
 
            text = str(r).strip() 
 
        #    skip duplicates 
        if text in seen: 
            continue 
 
        seen.add(text) 
 
        unique.append(text) 
 
    return unique

from app.storage.duckdb_store import ( 
    conn

def get_neighbor_chunks( 
 
    table_id, 
    row_index, 
    window=2 
 
): 
 
    min_row = max(0, row_index - window) 
 
    max_row = row_index + window 
 
    query = f""" 
 
    SELECT content 
    FROM engineering_chunks 
 
    WHERE table_id = '{table_id}' 
 
    AND CAST( 
 
        regexp_extract( 
            chunk_id, 
            'row_(\d+)' , 
            1 
 
        ) AS INTEGER 
 
    ) 
 
    BETWEEN {min_row} AND {max_row} 
 
    """ 
 
    rows = conn.execute(query).fetchall() 
 
    return [r[0] for r in rows]

from sentence_transformers import ( 
    CrossEncoder 
) 
 
reranker = CrossEncoder( 
    "BAAI/bge-reranker-base" 
) 
 
def rerank_results(query, documents): 
 
    pairs = [] 
 
    for doc in documents: 
        pairs.append([ 
            query, 
            doc["document"] 
        ]) 
 
    scores = reranker.predict(pairs) 
 
    ranked = [] 
 
    for doc, score in zip(documents, scores): 
 
        doc["rerank_score"] = float(score) 
 
        ranked.append(doc) 
 
    ranked = sorted( 
        ranked, 
        key=lambda x: x["rerank_score"], 
        reverse=True 
    ) 
 
    return ranked

def hybrid_search(query): 
 
    semantic = semantic_search( 
        query, 
        top_k=10 
    ) 
 
    semantic_docs = [] 
 
    docs = semantic["documents"][0] 
    metas = semantic["metadatas"][0] 
 
    for doc, meta in zip(docs, metas): 
 
        semantic_docs.append({ 
            "document": doc, 
            "metadata": meta, 
            "source": "semantic" 
        }) 
 
    keyword_docs = keyword_search(query) 
 
    combined = semantic_docs + keyword_docs 
 
    #    deduplicate 
    unique = {} 
 
    for item in combined: 
        unique[item["document"]] = item 
 
    final_docs = list(unique.values()) 
 
    #    rerank 
    reranked = rerank_results( 
        query, 
        final_docs 
    ) 
 
    return reranked[:5]

import fitz  # PyMuPDF 
 
def parse_pymupdf(file_path): 
    doc = fitz.open(file_path) 
 
    titles = [] 
    text = [] 
 
    for page_num, page in enumerate(doc): 
        content = page.get_text() 
 
        if not content: 
            continue 
 
        lines = content.split("\n") 
 
        for line in lines: 
            line = line.strip() 
 
            if not line: 
                continue 
 
            #    Title detection (important) 
            if ( 
                line.isupper() or 
                "TEST REPORT" in line or 
                "BRAKE" in line 
            ): 
                titles.append(line) 
            else: 
                text.append(line) 
 
    return { 
        "titles": titles, 
        "text": text 
    }

import pdfplumber 
 
def clean_row(row): 
    return [ 
        cell.strip() 
        for cell in row 
        if cell not in (None, "" , " ") 
    ] 
 
def extract_tables_pdf(file_path): 
    tables = [] 
 
    with pdfplumber.open(file_path) as pdf: 
 
        for page_num, page in enumerate(pdf.pages): 
 
            raw_tables = page.extract_tables() 
 
            if not raw_tables: 
                continue 
 
            for table_id, table in enumerate(raw_tables): 
 
                cleaned_table = [] 
 
                for row in table: 
                    cleaned = clean_row(row) 
 
                    if len(cleaned) >= 2: 
                        cleaned_table.append(cleaned) 
 
                if len(cleaned_table) < 2: 
                    continue 
 
                tables.append({ 
                    "page": page_num, 
                    "table_id": f"page_{page_num}table{table_id}" , 
                    "data": cleaned_table 
                }) 
 
                print(f"   Table extracted from page {page_num}") 
 
    return tables

import pandas as pd 
 
def get_row_profile(row): 
 
    values = [ 
        str(x).strip() 
        for x in row 
    ] 
 
    numeric_count = 0 
    text_count = 0 
    empty_count = 0 
 
    for v in values: 
 
        if v.lower() in ["" , "nan" , "none"]: 
            empty_count += 1 
 
        elif any(c.isdigit() for c in v): 
            numeric_count += 1 
 
        else: 
            text_count += 1 
 
    populated = len(values) - empty_count 
 
    return { 
        "numeric": numeric_count, 
        "text": text_count, 
        "empty": empty_count, 
        "populated": populated 
    }

def segment_dataframe(df): 
 
    segments = [] 
 
    current = [] 
 
    previous_profile = None 
 
    for _, row in df.iterrows(): 
 
        row_values = list(row) 
 
        profile = get_row_profile(row_values) 
 
        #    detect structural discontinuity 
        split_condition = False 
 
        if previous_profile: 
 
            diff = abs( 
                profile["populated"] - 
                previous_profile["populated"] 
            ) 
 
            #    sudden structure shift 
            if diff >= 3: 
                split_condition = True 
 
        if split_condition and current: 
 
            segments.append( 
                pd.DataFrame(current) 
            ) 
 
            current = [] 
 
        current.append(row_values) 
 
        previous_profile = profile 
 
    if current: 
        segments.append( 
            pd.DataFrame(current) 
        ) 
 
    return segments

        val_str = str(val).strip() 
 
        #    semantic parent row 
        if ( 
            val_str.lower() not in 
            ["" , "nan" , "none"] 
        ): 
 
            previous_value = val 
 
            propagated.append(val) 
 
        else: 
 
            propagated.append(previous_value) 
 
    df.iloc[:, 0] = propagated 
 
    return df

import pandas as pd 
 
def filter_useful_tables(tables): 
 
    useful = [] 
 
    for t in tables: 
 
        df = t["df"].copy() 
 
        #    normalize empty values 
        df = df.map( 
            lambda x: pd.NA 
            if str(x).strip().lower() 
            in ["nan" , "none" , ""] 
            else x 
        ) 
 
        #    remove fully empty columns 
        df = df.dropna(axis=1, how="all") 
 
        #    remove fully empty rows 
        df = df.dropna(axis=0, how="all") 
 
        rows, cols = df.shape 
 
        #    skip useless tiny tables 
        if rows <= 1: 
            continue 
 
        if cols <= 1: 
            continue 
 
        #    update cleaned dataframe 
        t["df"] = df 
 
        useful.append(t) 
 
    return useful

def dataframe_to_chunks(table): 
 
    chunks = [] 
 
    df = table["df"] 
 
    page = table["page"] 
 
    table_id = table["table_id"] 
 
    for idx, row in df.iterrows(): 
 
        values = [ 
            str(v) 
            for v in row.values 
        ] 
 
        text = " | " .join(values) 
 
        chunks.append({ 
            "id": f"{table_id}row{idx}" , 
            "text": text, 
            "metadata": { 
                "page": page, 
                "table_id": table_id 
            } 
        }) 
 
    return chunks

from app.memory.session_memory import ( 
    get_history 
) 
 
def rewrite_query(current_query): 
 
    history = get_history() 
 
    #    previous user queries 
    previous_queries = [ 
 
        h["content"] 
 
        for h in history 
 
        if h["role"] == "user" 
    ] 
 
    #    no history 
    if not previous_queries: 
 
        return current_query 
 
    previous_query = previous_queries[-1] 
 
    #    avoid duplicate expansion 
    if current_query.lower() in previous_query.lower(): 
 
        return current_query 
 
    #    contextual rewrite 
    rewritten = ( 
        previous_query 
        + " " 
        + current_query 
    ) 
 
    return rewritten

def deduplicate_context(results): 
 
    unique = [] 
 
    seen = set() 
 
    for r in results: 
 
        if isinstance(r, dict): 
 
            text = str( 
                r.get("document" , "") 
            ).strip() 
 
        else: 
 
            text = str(r).strip() 
 
        #    skip duplicates 
        if text in seen: 
            continue 
 
        seen.add(text) 
 
        unique.append(text) 
 
    return unique

def get_neighbor_chunks( 
 
    table_id, 
    row_index, 
    window=2 
 
): 
 
    min_row = max(0, row_index - window) 
 
    max_row = row_index + window 
 
    query = f""" 
 
    SELECT content 
    FROM engineering_chunks 
 
    WHERE table_id = '{table_id}' 
 
    AND CAST( 
 
        regexp_extract( 
            chunk_id, 
            'row_(\d+)' , 
            1 
 
        ) AS INTEGER 
 
    ) 
 
    BETWEEN {min_row} AND {max_row} 
 
    """ 
 
    rows = conn.execute(query).fetchall() 
 
    return [r[0] for r in rows]

from sentence_transformers import ( 
    CrossEncoder 
) 
 
reranker = CrossEncoder( 
    "BAAI/bge-reranker-base" 
) 
 
def rerank_results(query, documents): 
 
    pairs = [] 
 
    for doc in documents: 
        pairs.append([ 
            query, 
            doc["document"] 
        ]) 
 
    scores = reranker.predict(pairs) 
 
    ranked = [] 
 
    for doc, score in zip(documents, scores): 
 
        doc["rerank_score"] = float(score) 
 
        ranked.append(doc) 
 
    ranked = sorted( 
        ranked, 
        key=lambda x: x["rerank_score"], 
        reverse=True 
    ) 
 
    return ranked
